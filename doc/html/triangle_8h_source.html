Drone Project
This project models a network of drones and servers within a geometric environment where communication and movement are based on a graph structure. The primary objective is to simulate drone routing and interaction with servers, subject to spatial constraints and communication rules. Drones are tasked with navigating a network of servers, and only the first drone to arrive at each server will continue its route. The others will wait their turn, landing at the server until it is their turn to proceed.

326 void checkIsOpen(QVector<Triangle*> triangles) {
327
328 // Boolean flags to track if the edges are shared with other triangles
329 bool edge01Shared = false;
330 bool edge02Shared = false;
331 bool edge12Shared = false;
332
333 // Loop through all triangles to check if edge 0-1 is shared
334 int i = 0;
335 while (i < triangles.size() && !edge01Shared) {
336 if (triangles[i] != this) { // Ensure we're not checking against the current triangle
337 // Check if edge 0-1 or 1-0 is shared with another triangle
338 if (triangles[i]->hasEdge(this->getVertexPtr(0), this->getVertexPtr(1)) ||
339 triangles[i]->hasEdge(this->getVertexPtr(1), this->getVertexPtr(0))) {
340 edge01Shared = true; // Mark this edge as shared
341 }
342 }
343 ++i; // Move to the next triangle
344 }
345
346 // Reset the counter and check for edge 0-2
347 i = 0;
348 while (i < triangles.size() && !edge02Shared) {
349 if (triangles[i] != this) {
350 // Check if edge 0-2 or 2-0 is shared with another triangle
351 if (triangles[i]->hasEdge(this->getVertexPtr(0), this->getVertexPtr(2)) ||
352 triangles[i]->hasEdge(this->getVertexPtr(2), this->getVertexPtr(0))) {
353 edge02Shared = true; // Mark this edge as shared
354 }
355 }
356 ++i;
357 }
358
359 // Reset the counter and check for edge 1-2
360 i = 0;
361 while (i < triangles.size() && !edge12Shared) {
362 if (triangles[i] != this) {
363 // Check if edge 1-2 or 2-1 is shared with another triangle
364 if (triangles[i]->hasEdge(this->getVertexPtr(1), this->getVertexPtr(2)) ||
365 triangles[i]->hasEdge(this->getVertexPtr(2), this->getVertexPtr(1))) {
366 edge12Shared = true; // Mark this edge as shared
367 }
368 }
369 ++i;
370 }
371
372 // A triangle is open if at least one of its edges is not shared with other triangles
373 this->isOpen = !(edge01Shared && edge02Shared && edge12Shared);
374
375 // If the triangle is open, add its non-shared edges to the openSides set
376 if (this->isOpen) {
377 if (!edge01Shared) {
378 // Edge 0-1 is not shared, add it to openSides
379 openSides.insert(qMakePair(this->getVertexPtr(0), this->getVertexPtr(1)));
380 }
381 if (!edge02Shared) {
382 // Edge 0-2 is not shared, add it to openSides
383 openSides.insert(qMakePair(this->getVertexPtr(0), this->getVertexPtr(2)));
384 }
385 if (!edge12Shared) {
386 // Edge 1-2 is not shared, add it to openSides
387 openSides.insert(qMakePair(this->getVertexPtr(1), this->getVertexPtr(2)));
388 }
389 }
390 }
